\item $p \vee \neg (q \implies r) \text{ : the boolean formula }$

\item $p \vee \neg (\neg q \vee r) \text{ : by the equivalence to (1) } (a \implies b) \equiv (\neg a \vee b)$
\\
\item $p \vee (\neg \neg q \wedge \neg r) \text{ : by De Morgan's law to (2) } \neg (a \vee b) \equiv (\neg a \wedge \neg b)$
\\
\item $p \vee (q \wedge \neg r) \text{ : by  De Morgan's law to (3) } \neg \neg a \equiv a$
\\
\item $(p \vee q) \wedge (p \vee \neg r) \text{ : by distribution of } \wedge \text{over} \vee \text{to (4) } a \vee (b \wedge c) \equiv (a \vee b) \wedge (a \vee c) $
\\
\item $(p \vee q) \wedge (p \vee \neg r)$ is in CNF
\end{enumerate}
